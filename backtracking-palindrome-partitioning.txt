func partition(s string) [][]string {
    result := [][]string{}
    var backTracking func(path []string, remainingString string)
    backTracking = func(path []string, remainingString string){
        if len(remainingString) == 0{
            result = append(result,append([]string{},path...))
            return
        }

        for i:=1;i<=len(remainingString);i++{
            choice := remainingString[:i]//0:i
            if !isPalindrome(choice){
                continue
            }

            path = append(path,choice)
            backTracking(path,remainingString[i:])// i:len(remainingString)-1
            path = path[:len(path)-1]
        }
    }
    backTracking([]string{},s)
    return result
}

func isPalindrome(s string) bool{
    i := 0
    j := len(s)-1
    for i < j{
        if s[i] != s[j]{
            return false
        }
        i++
        j--
    }
    return true
}
