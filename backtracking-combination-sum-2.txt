// Time Complexity O(2^n)
// Space Complexity O(n)
func combinationSum2(candidates []int, target int) [][]int {
    result := [][]int{}
    sort.Ints(candidates)
    var backTracking func(path []int, start int, remainingSum int)
    backTracking = func(path []int, start int, remainingSum int){
        if remainingSum == 0{
            result = append(result,append([]int{},path...))
        }

        for i:=start;i<len(candidates);i++{
            if i > start && candidates[i-1] == candidates[i] || candidates[i] > remainingSum{
                continue
            }
            path = append(path,candidates[i])
            backTracking(path,i+1,remainingSum - candidates[i])
            path = path[:len(path)-1]
        }
    }
    backTracking([]int{},0,target)
    return result
}
