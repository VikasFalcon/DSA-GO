func nextGreaterElements(nums []int) []int {
    n := len(nums)
    result := make([]int,n)
    for i := range result{
        result[i] = -1
    }

    stack := []int{}
    for i := 2*n-1;i>=0;i--{
        idx := i%n
        for len(stack) > 0 && stack[len(stack)-1] <= nums[idx] {
            stack = stack[:len(stack)-1]
        }

        if len(stack) > 0 {
            result[idx] = stack[len(stack)-1]
        }

        stack = append(stack,nums[idx])
    }

    return result
}
