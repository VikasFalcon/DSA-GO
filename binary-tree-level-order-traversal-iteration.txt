/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
func levelOrder(root *TreeNode) [][]int {
    if root == nil {
        return [][]int{}
    }
    var curr *TreeNode
    queue := []*TreeNode{root}
    var ans [][]int
    for len(queue)>0 {
        levelSize:=len(queue)
        var levelSlice []int
        for i:=0;i<levelSize;i++{
            curr = queue[0]
            queue = queue[1:]

            levelSlice = append(levelSlice,curr.Val) 

            if curr.Left != nil {
                queue = append(queue, curr.Left)
            }

            if curr.Right != nil {    
                queue = append(queue, curr.Right)
            }

            
        }   
        ans = append(ans, levelSlice)
    }
    return ans
}
